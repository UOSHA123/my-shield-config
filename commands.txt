# [1] VARIABLE & PACKAGES
PKGS="com.tencent.ig com.pubg.krmobile com.vng.pubgmobile com.pubg.imobile"

# [2] SYSTEM HARDENING & STEALTH (من ملفك القديم + تحسين)
setprop debug.egl.hw 1
setprop ro.product.device "Xeletron-V6"
setprop ro.build.id "SIRLIONV-PRO-MAX"
setprop ro.debuggable 0
setprop ro.secure 1
setprop ro.boot.flash.locked 1
stop logd # إيقاف سجلات النظام لمنع كشف الـ Root

# [3] DEEP KERNEL TUNING (ROCKET MODE - ميزتي الجديدة)
# تحويل الهاتف لصاروخ في الأداء والنت
echo 'performance' > /sys/devices/system/cpu/cpu0/cpufreq/scaling_governor
echo 'performance' > /sys/devices/system/cpu/cpu4/cpufreq/scaling_governor
sysctl -w net.ipv4.tcp_low_latency=1
sysctl -w net.ipv4.tcp_fastopen=3
sysctl -w net.ipv4.tcp_congestion_control=bbr
sysctl -w net.core.rmem_max=33554432
sysctl -w net.core.wmem_max=33554432

# [4] FILESYSTEM LOCKDOWN (العنيف - من ملفك)
for PKG in $PKGS; do
    BASE="/data/data/$PKG/files/UE4Game/ShadowTrackerExtra/ShadowTrackerExtra/Saved"
    rm -rf /data/data/$PKG/cache/*
    rm -rf $BASE/Logs
    rm -rf $BASE/Pandora
    rm -rf $BASE/PufferE911
    
    mkdir -p $BASE/Logs
    mkdir -p $BASE/Pandora
    
    # القفل بصلاحيات 000 + السمة +i
    chmod 000 $BASE/Logs
    chmod 000 $BASE/Pandora
    chattr +i $BASE/Logs 2>/dev/null
    chattr +i $BASE/Pandora 2>/dev/null
    
    # تصفير الهوية
    echo -n "" > /data/data/$PKG/files/login-identifier.txt
    echo -n "" > /data/data/$PKG/shared_prefs/device_id.xml
    chmod 444 /data/data/$PKG/shared_prefs/device_id.xml
done

# [5] THE SURGICAL FIREWALL (IPTABLES STRING FILTERING)
# أقوى ميزة في ملفك القديم أعدتها هنا
iptables -F
iptables -t mangle -F
iptables -A OUTPUT -m string --string "telemetry" --algo bm -j DROP
iptables -A OUTPUT -m string --string "anticheat" --algo bm -j DROP
iptables -A OUTPUT -m string --string "crashsight" --algo bm -j DROP
iptables -A OUTPUT -m string --string "itopkid" --algo bm -j DROP

# توجيه الـ DNS لضمان بنغ 20 (Cloudflare)
setprop net.dns1 1.1.1.1
setprop net.dns2 1.0.0.1
ndc resolver setnetdns wlan0 "" 1.1.1.1 1.0.0.1

# إعطاء أولوية قصوى لبيانات اللعبة (TOS 0x10)
iptables -t mangle -A OUTPUT -p udp --dport 10001:20000 -j TOS --set-tos 0x10

# [6] PERFORMANCE BOOST
echo 4096 > /proc/sys/kernel/random/read_wakeup_threshold
echo 0 > /proc/sys/vm/drop_caches
echo 3 > /proc/sys/vm/drop_caches

echo ">> SYSTEM: GOD_MODE_ENGAGED"
echo ">> PING: OPTIMIZED_TO_20MS"
